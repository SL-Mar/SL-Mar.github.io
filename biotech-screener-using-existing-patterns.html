    // Calculate RSI(14) function calculateRSI(prices, period = 14) { let gains = 0, losses = 0;

    for (let i = 1; i <= period; i++) { const change = prices[i] - prices[i-1]; if (change > 0) gains += change; else losses += Math.abs(change); }

    const avgGain = gains / period; const avgLoss = losses / period; const rs = avgGain / avgLoss; return 100 - (100 / (1 + rs)); }

    const closePrices = historicalData.map(d => d.close); const rsi = calculateRSI(closePrices);

    // Calculate 20-day and 50-day MA const ma20 = closePrices.slice(-20).reduce((a,b) => a+b, 0) / 20; const ma50 = closePrices.slice(-50).reduce((a,b) => a+b, 0) / 50;

    return { json: { ...$json, rsi: rsi, ma20: ma20, ma50: ma50, priceAboveMA20: $json.close > ma20, priceAboveMA50: $json.close > ma50 } };
